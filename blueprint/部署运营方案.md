### 2.2 迁移脚本设计

```python
# app/migration/from_obsidian.py (伪代码)

class ObsidianMigrator:
    """将 Obsidian 知识库迁移到 PostgreSQL"""
    
    def __init__(self, source_dir: str, db_session):
        self.source_dir = Path(source_dir)
        self.db = db_session
    
    def migrate(self):
        notes = self.scan_markdown_files()
        
        for note_path in notes:
            # 1. 解析 frontmatter 和正文
            metadata, content = self.parse_frontmatter(note_path)
            
            # 2. 提取 wikilink
            wikilinks = self.extract_wikilinks(content)
            
            # 3. 确定分类
            category = self.infer_category(note_path, metadata)
            
            # 4. 写入数据库
            note = Note(
                title=metadata.get('title', note_path.stem),
                content=content,
                frontmatter=metadata,
                category=category,
                tags=metadata.get('tags', []),
                status='published',
                author_id=self.get_or_create_author(metadata),
                created_at=metadata.get('created'),
                source_path=str(note_path)
            )
            self.db.add(note)
        
        self.db.commit()
        
        # 5. 建立链接关系
        self.build_link_graph()
        
        # 6. 生成向量嵌入
        self.generate_embeddings()
    
    def parse_frontmatter(self, path: Path):
        """解析 YAML frontmatter"""
        content = path.read_text(encoding='utf-8')
        match = re.match(r'^---\n(.*?)\n---\n(.*)', content, re.DOTALL)
        if match:
            metadata = yaml.safe_load(match.group(1))
            body = match.group(2)
        else:
            metadata = {}
            body = content
        return metadata, body
    
    def extract_wikilinks(self, content: str) -> list:
        """提取 [[笔记名]] 和 [[路径|别名]]"""
        pattern = r'\[\[([^\]|]+)(?:\|([^\]]+))?\]\]'
        return re.findall(pattern, content)
    
    def infer_category(self, path: Path, metadata: dict) -> str:
        """从路径推断分类"""
        # D:\WenShanKB\01-地理与自然环境\xxx.md → 01-地理与自然环境
        relative = path.relative_to(self.source_dir)
        if len(relative.parts) > 1:
            return relative.parts[0]
        return metadata.get('category', '未分类')
```

## 三、CI/CD 流水线

```yaml
# .github/workflows/deploy.yml
name: Deploy

on:
  push:
    branches: [main]
  workflow_dispatch:  # 手动触发

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Run tests
        run: |
          cd api && pytest
          cd web && npm test

  build-and-deploy:
    needs: test
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      
      - name: Build Docker images
        run: |
          docker compose build
      
      - name: Push to registry
        run: |
          docker tag wenshan-kb-api registry.example.com/wenshan-kb-api:${{ github.sha }}
          docker push registry.example.com/wenshan-kb-api:${{ github.sha }}
      
      - name: Deploy to server
        uses: appleboy/ssh-action@v1
        with:
          host: ${{ secrets.SERVER_HOST }}
          username: deploy
          key: ${{ secrets.SSH_KEY }}
          script: |
            cd /opt/wenshan-kb
            docker compose pull
            docker compose up -d --remove-orphans
            docker compose exec api python -m app.db.migrate
```
